//Variables
  String name = "Saf";
  var x = 41;
  dynamic firstName = "Safventure";
  const String fullName = "SafII";
  final String nickName = "J";

  print(x);
  print(name);
  print(firstName);
  print (fullName);
  print(nickName);

//Data Types
  Strings, Integers, Doubles, Booleans,
  Dynamic- if you haven't decided what datatype

//List
  var myList = [1,2,3];
  print(myList);
  print(myList[0]);
  
//change an item
  myList[0] = 41;
  print(myList);
  
//create an empty list
  var emptyList = [];
  print(emptyList);
  
//add empty list
  emptyList.add(41);
  print(emptyList);
  
//add multiple to empty list
  emptyList.addAll([1,2,3]);
  print(emptyList);
  
//insert at specific position 
  myList.insert(3,900);
  print(myList);
  
//insert many
  myList.insertAll(1,[99,98,97]);
  print(myList);
  
//mixed lists
  var mixedList = [1,2,3, "John", "Bob"];
  print(mixedList);
  
//remove from list
  mixedList.remove("John");
  print(mixedList);
  
//remove from specific location
  mixedList.removeAt(1);
  print(mixedList);
